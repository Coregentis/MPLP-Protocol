interface DesignPrinciples {
  // 1. 简单易用 - 降低学习成本
  simplicity: {
    principle: "复杂的AI能力，简单的用户交互",

    layout: "三栏布局 - 侧边栏 + 主内容 + 辅助面板",
    features: "完整功能，多窗口支持"
  };
  
  // 平板端
  tablet: {
    width: "768px - 1023px", 
    layout: "两栏布局 - 可折叠侧边栏 + 主内容",
    features: "核心功能，简化界面"
  };
  
  // 手机端 (查看和简单操作)
  mobile: {
    maxWidth: "767px",
    layout: "单栏布局 - 底部导航 + 全屏内容",

  };
  
  // 自适应策略
  adaptiveStrategy: {
    contentPriority: "根据屏幕大小调整内容优先级",

    interactionMode: "桌面端鼠标 → 平板端触摸 → 手机端手势"
  };
